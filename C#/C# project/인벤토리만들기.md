## 추가할 기능
- 아이템 중첩

### 만드는 과정에서의 고민 ###
- List가 아닌 Array인 이유
  1. 최대용량이 정해져있고, 비어있는 공간을 null로 채워놓기 편함.
  2. 빈공간(null), 비어있지 않은 공간을 index를 통해 자유롭게 접근하기 용이함.

- 입력을 받는 메서드(InputIndex), 로직의 적합한 위치?
  -> 지금은 규모가 작은 프로그램이기도 하고, Inventory 클래스에 넣어놓는게 편해서 그렇게 했지만, 사실 입력처리는 static 클래스나 Main에서 처리하는게 맞을듯

- 방어적 코딩
  -> 처음으로 신경써서 짜보기 시작했는데 앞으로도 습관을 들이면 좋을듯

- 정렬 구현 과정
  -> 그냥 Array.Sort()쓰려고 했는데 클래스기도 하고 배열에 null도 섞여있어서 커스텀을 사용해야함, 방법 찾는 과정에서 CompareTo 원리, switch식을 알게됨

```csharp
using System;

class Item  // 아이템 클래스
{
	public string Name{get; set;} // 아이템 이름
	public string Symbol{get;set;} // 아이템 아이콘
	public int Gold{get; set;} // 아이템 가격
	public Item(string _name, int _gold,string _symbol) // 멤버변수를 초기화해야만 객체 생성 가능
	{
		Name=_name;
		Gold=_gold;
		Symbol=_symbol;
	}
	public void ShowItemInfo() // 아이템 정보 출력
	{
		Console.WriteLine($"이름 : {Name}");
		Console.WriteLine($"가격 : {Gold}G");
		Console.WriteLine($"심볼 : {Symbol}");
	}
}


class Inventory
{
	Item[] ArrItem; // 아이템을 저장할 배열
	int rows; // 아이템 크기 행
	int cols; // 아이템 크기 열
	public Inventory(int x,int y)   // 인벤토리의 크기를 정해야만 객체 생성 가능
	{
		if(x<1)
			x=1;
		if(y<1)
			y=1;
		rows=y;
		cols=x;
		ArrItem = new Item[x*y];
	}
	
	public void Render()   // 인벤토리 형태 출력
	{
		for(int i=0;i<ArrItem.Length;i++)
		{
			if(i%cols==0)
				Console.WriteLine("");
			if(ArrItem[i] == null)
				Console.Write("□");
			else
				Console.Write(ArrItem[i].Symbol);
		}
	}
	public void ShowOption()  // 인벤토리 유틸리티 접근 목록 출력
	{
		Console.WriteLine("");
		Console.WriteLine("1. 아이템 조회");
		Console.WriteLine("2. 아이템 추가");
		Console.WriteLine("3. 아이템 삭제");
		Console.WriteLine("4. 아이템 이동");
		Console.WriteLine("5. 아이템 정렬");
		Console.WriteLine("");
	}
	
	private int InputIndex(string message){   // 각 유틸리티에 사용될 인덱스 입력 처리 메서드
		int _x;
		int _y;
		while(true)
		{
			Console.WriteLine(message); 
			Console.WriteLine("몇행?");
			string input1 = Console.ReadLine();
			if(int.TryParse(input1,out _x)&&_x<rows&&_x>=0){	
			}
			else
			{
				Console.WriteLine("잘못된 입력입니다.");
				continue;
			}
			
			Console.WriteLine("몇열?");
			string input2 = Console.ReadLine();
			if(int.TryParse(input2,out _y)&&_y<cols&&_y>=0){	
				break;
			}
			else
			{
				Console.WriteLine("잘못된 입력입니다.");
				continue;
			}
		}
		
		int index = _x*cols+_y;
		return index;     
	}
	
	public void ShowItem()
	{	
		int index = InputIndex("-- 조회할 아이템 입력 --");
		if(ArrItem[index]!=null)       // 비어있지 않은 경우 아이템 정보 출력
		{
			ArrItem[index].ShowItemInfo();
			Console.ReadLine();
		}
		else
		{
			Console.WriteLine("비어있음!");   // 비어있는 경우 실패
		}
	}
	public void AddItem(Item _item)
	{
		for(int i=0;i<ArrItem.Length;i++){
			if(ArrItem[i]==null)
			{
				ArrItem[i]=_item;
				return;
			}
		}
		Console.WriteLine("가방이 가득찼습니다!");
		return;
	}
	
	public void RemoveItem()
	{
		
		int index = InputIndex("-- 삭제할 아이템 입력 --");
		
		if(ArrItem[index]!=null)   // 지정한 인덱스에 아이템이 있는 경우 삭제
		{
			ArrItem[index]=null;
			Console.WriteLine("삭제성공!");
			Console.ReadLine();
		}
		else
		{
			Console.WriteLine("이미 비어있음!");  // 지정한 인덱스에 아이템이 없는 경우 실패
		}
	}
	
	public void MoveItem()
	{
		int fromIndex;
		while(true)
		{
			fromIndex = InputIndex("-- 옮길 아이템 입력 --");
			if(ArrItem[fromIndex]!=null)
			{
				break;
			}
			Console.WriteLine("잘못된 지정!");
		}
		int toIndex = InputIndex(">> 옮길 위치 입력 <<");
	
		Item temp = ArrItem[fromIndex];
		ArrItem[fromIndex] = ArrItem[toIndex];
		ArrItem[toIndex] = temp;
	}
	public void SortItem()
	{
		Console.WriteLine("1. 이름 기준 정렬");
		Console.WriteLine("2. 가격 기준 정렬");
		string _option = Console.ReadLine();
		
		/*switch(_option)
		{
			case "1" :
				Array.Sort(ArrItem, (A1,A2) =>
						   {
							   if(A1==null&&A2==null) return 0;
							   if(A1==null)return 1;
							   if(A2==null)return -1;
							   return A1.Name.CompareTo(A2.Name);
							});
				break;
			case "2" :
				Array.Sort(ArrItem, (A1,A2)=>
						   {
							   if(A1==null&&A2==null) return 0;
							   if(A1==null) return 1;
							   if(A2==null) return -1;
							   return A1.Gold.CompareTo(A2.Gold);
							});
				break;
			default :
				Console.WriteLine("잘못된 입력");
				break;
		}*/
		Array.Sort(ArrItem, (A1,A2) => // CompareTo=>양수일때 앞 요소를 뒤로 넘기는 구조
		{
			if(A1==A2) return 0;
			if(A1==null) return 1;
			if(A2==null) return -1;

			return _option switch  // switch식
			{
				"1" => A1.Name.CompareTo(A2.Name),
				"2" => A1.Gold.CompareTo(A2.Gold),
				_=> 0
			};
		});
	}
}


public class Program
{
	public static void Main()
	{
		Inventory Inven = new Inventory(5,3);
		Inven.AddItem(new Item("철",100,"■"));
		Inven.AddItem(new Item("가죽",100,"★"));
		Inven.AddItem(new Item("장갑",100,"◆"));
		
		while(true)
		{
			Inven.Render();
			Inven.ShowOption();
			string option = Console.ReadLine();
			
			switch(option)
			{
				case "1" :
					Inven.ShowItem();
					break;
				case "3" :
					Inven.RemoveItem();
					break;
				case "4" :
					Inven.MoveItem();
					break;
				case "5" :
					Inven.SortItem();
					break;
				default :
					Console.WriteLine("잘못된 입력");
					break;
			}
		}
	}
}
```
