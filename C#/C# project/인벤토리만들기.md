## 추가할 기능
- 아이템 중첩
- 아이템 이동
- 아이템 정렬

```csharp
using System;

class Item
{
	public string Name{get; set;}
	public string Symbol{get;set;}
	public int Gold{get; set;}
	public Item(string _name, int _gold,string _symbol)
	{
		Name=_name;
		Gold=_gold;
		Symbol=_symbol;
	}
}

class Inventory
{
	Item[] ArrItem;
	int rows;
	public Inventory(int x,int y)
	{
		if(x<1)
			x=1;
		if(y<1)
			y=1;
		rows=x;
		ArrItem = new Item[x*y];
	}
	
	public void Render()
	{
		for(int i=0;i<ArrItem.Length;i++)
		{
			if(i%rows==0)
				Console.WriteLine("");
			if(ArrItem[i] == null)
				Console.Write("□");
			else
				Console.Write(ArrItem[i].Symbol);
		}
	}
	
	public void AddItem(Item _item)
	{
		for(int i=0;i<ArrItem.Length;i++){
			if(ArrItem[i]==null)
			{
				ArrItem[i]=_item;
				return;
			}
		}
	}
	
	public void RemoveItem(int _x, int _y)
	{
		int target=rows*_x+_y;
		if(target>=ArrItem.Length||target<0)
			Console.WriteLine("잘못된 범위지정");
		else
			ArrItem[target]=null;
	}
}


public class Program
{
	public static void Main()
	{
		Inventory Inven = new Inventory(5,3);
		Inven.AddItem(new Item("철",100,"■"));
		Inven.AddItem(new Item("가죽",100,"★"));
		Inven.AddItem(new Item("장갑",100,"◆"));
		Inven.RemoveItem(0,0);
		
		while(true)
		{
			Inven.Render();
			Console.ReadLine();
			
		}
	}
}
```
